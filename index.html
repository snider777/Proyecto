import javax.swing.*; import java.awt.*; import java.awt.event.*; public class ImagenConBoton extends JFrame { public ImagenConBoton() { setTitle("Imagen con Botón EMPEZAR"); setSize(800, 600); setDefaultCloseOperation(EXIT_ON_CLOSE); setLocationRelativeTo(null); // Centrar en pantalla // Panel que contiene imagen de fondo y botón FondoPanel fondo = new FondoPanel("mundo.jpg"); fondo.setLayout(new GridBagLayout()); // Para centrar el botón JButton boton = new JButton("EMPEZAR"); boton.setFont(new Font("Arial", Font.BOLD, 20)); fondo.add(boton); // Se ubicará en el centro gracias a GridBagLayout add(fondo); setVisible(true); } // Clase interna que dibuja la imagen como fondo class FondoPanel extends JPanel { private Image imagen; public FondoPanel(String archivoImagen) { imagen = new ImageIcon(archivoImagen).getImage(); } @Override protected void paintComponent(Graphics g) { super.paintComponent(g); // Dibuja la imagen centrada y escalada al tamaño del panel g.drawImage(imagen, 0, 0, getWidth(), getHeight(), this); } } public static void main(String[] args) { new ImagenConBoton(); } }
